еҸ‘иЎЁдәҺ 2016-07-19 | еҲҶзұ»дәҺ iOS | | йҳ…иҜ»ж¬Ўж•°:

iOSи®ҫеӨҮдёҠеӨ§еӨҡж•°еҗҺзҪ®ж‘„еғҸеӨҙйғҪж”ҜжҢҒеҹәдәҺз»ҷе®ҡе…ҙи¶ЈзӮ№и®ҫзҪ®еҜ№з„Ұе’Ңжӣқе…үж•°жҚ®.еңЁз•ҢйқўдёҠзҡ„зӣҙжҺҘдҪ“зҺ°е°ұжҳҜзӮ№еҮ»и§Ҷйў‘йў„и§Ҳз•Ңйқўзҡ„жҹҗдёӘдҪҚзҪ®,е°ұдјҡеңЁиҝҷдёӘзӮ№иҮӘеҠЁеҜ№з„Ұе’Ңжӣқе…ү.жң¬ж–Үе°Ҷе®һзҺ°зӮ№еҮ»еҜ№з„Ұе’ҢзӮ№еҮ»жӣқе…үеҠҹиғҪ.

Kcameraзҡ„д»Јз ҒеҸҜд»ҘеңЁhttps://github.com/changjianfeishui/KcameraдёҠжүҫеҲ°.

зӮ№еҮ»еҜ№з„ҰеҠҹиғҪзҡ„е®һзҺ°

еқҗж Үз©әй—ҙиҪ¬жҚў

еңЁiOSи®ҫеӨҮдёӯ,4еҜёеұҸзҡ„еұҸе№•еқҗж Үе·ҰдёҠи§’дёә(0,0),еһӮзӣҙжЁЎејҸдёӢеҸідёӢи§’еқҗж Үдёә(320,568),ж°ҙе№іжЁЎејҸдёӢеҸідёӢи§’еқҗж Үдёә(568,320).иҖҢжҚ•жҚүи®ҫеӨҮеқҗж Үзі»еҲҷдёҚеҗҢ,и®ҫеӨҮе·ҰдёҠи§’дёә(0,0),еҸідёӢи§’дёә(1,1),е№¶дё”дёҚж”ҜжҢҒж°ҙе№іж–№еҗ‘ж—ӢиҪ¬.

еңЁPreView.swiftдёӯеўһеҠ еҰӮдёӢж–№жі•:

//е°ҶеұҸе№•еқҗж Үзі»дёҠзҡ„и§ҰжҺ§зӮ№иҪ¬жҚўдёәж‘„еғҸеӨҙи®ҫеӨҮеқҗж Үзі»дёҠзҡ„зӮ№
func captureDevicePointForPoint(point:CGPoint) -> CGPoint{
    let layer = self.layer as! AVCaptureVideoPreviewLayer
    return layer.captureDevicePointOfInterestForPoint(point)
}

override func pointInside(point: CGPoint, withEvent event: UIEvent?) -> Bool {
    if self.statusView.pointInside(self.convertPoint(point, toView: self.statusView), withEvent: event) || self.modeView.pointInside(self.convertPoint(point, toView: self.modeView), withEvent: event){

//зҹ©еҪўжЎҶзј©ж”ҫеҠЁз”»
func runBoxAnimationOnView(view:UIView,point:CGPoint) -> Void {
    view.center = point
    view.hidden = false
    UIView.animateWithDuration(0.15, delay: 0, options: .CurveEaseInOut, animations: { 
        view.layer.transform = CATransform3DMakeScale(0.5, 0.5, 1)
        }) { (_) in
            let time = dispatch_time(DISPATCH_TIME_NOW, Int64(0.5 * Double(NSEC_PER_SEC)))
            dispatch_after(time, dispatch_get_main_queue(), { 
                view.hidden = true
                view.layer.transform = CATransform3DIdentity
            })


    }
}

func focusAtPoint(point:CGPoint) -> Void {
    //1. иҺ·еҸ–еҪ“еүҚдҪҝз”Ёзҡ„ж‘„еғҸеӨҙ
    let device = self.activeVideoInput.device
    //2. еҲӨж–ӯеҪ“еүҚи®ҫеӨҮжҳҜеҗҰж”ҜжҢҒе…ҙи¶ЈзӮ№еҜ№з„Ұе’ҢиҮӘеҠЁеҜ№з„ҰжЁЎејҸ
    /*жҜ”еҰӮеүҚзҪ®ж‘„еғҸеӨҙе°ұдёҚж”ҜжҢҒеҜ№з„Ұж“ҚдҪң,еӣ дёәи®ҫеӨҮе’Ңзӣ®ж Үзҡ„и·қзҰ»дёҚдјҡеӨӘй•ҝ*/
    if device.focusPointOfInterestSupported && device.isFocusModeSupported(.AutoFocus) {
        //3. дҝ®ж”№й…ҚзҪ®еүҚйңҖиҰҒе…Ҳй”Ғе®ҡи®ҫеӨҮ(и®ҫеӨҮжҳҜеӨҡдёӘеә”з”ЁзЁӢеәҸйҖҡз”Ёзҡ„)
        if ((try? device.lockForConfiguration()) != nil) {
            //4. и®ҫзҪ®еҜ№з„ҰзӮ№,дҝ®ж”№еҜ№з„ҰжЁЎејҸ
            device.focusPointOfInterest = point
            device.focusMode = .AutoFocus
            device.unlockForConfiguration()
        }
    }else{
        //жҡӮж—¶еҝҪз•ҘдәҶеҜ№и®ҫеӨҮдёҚж”ҜжҢҒжғ…еҶөзҡ„еӨ„зҗҶ,жҜ”еҰӮеүҚзҪ®ж‘„еғҸеӨҙ
    }

}

func handDoubledoubleTap(tap:UITapGestureRecognizer) -> Void {
    //1. еҠЁз”»
    let center = (self.layer as! AVCaptureVideoPreviewLayer).pointForCaptureDevicePointOfInterest(CGPoint(x: 0.5, y: 0.5))
    self.focusBox.center = center
    self.exposureBox.center = center
    self.exposureBox.transform = CGAffineTransformMakeScale(1.2, 1.2)
    self.focusBox.hidden = false
    self.exposureBox.hidden = false
    UIView.animateWithDuration(0.15, delay: 0, options:.CurveEaseInOut, animations: { 
        self.focusBox.layer.transform = CATransform3DMakeScale(0.5, 0.5, 1.0)
        self.exposureBox.layer.transform = CATransform3DMakeScale(0.7, 0.7, 1.0)
        }) { (_) in
            let time = dispatch_time(DISPATCH_TIME_NOW, Int64(0.5 * Double(NSEC_PER_SEC)))
            dispatch_after(time, dispatch_get_main_queue(), {
                self.focusBox.hidden = true
                self.exposureBox.hidden = true
                self.focusBox.transform = CGAffineTransformIdentity
                self.exposureBox.transform = CGAffineTransformIdentity
            })
    }
    //2. йҖҡзҹҘд»ЈзҗҶ
